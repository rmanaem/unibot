PREFIX vivo: <http://vivoweb.org/ontology/core#>
PREFIX bibo: <http://purl.org/ontology/bibo/>
           
SELECT ?title ?subjectArea  
   WHERE{
     ?x rdf:type vivo:Course.
     ?x vivo:hasSubjectArea ?subjectArea.
     ?x vivo:Title ?title.
     ?x vivo:hasSubjectArea ?subjectCode.
     ?x vivo:Catalog ?courseNum.
     ?x vivo:description ?courseDescription.
     FILTER (regex(?courseDescription, 'anthropology')).
   }
